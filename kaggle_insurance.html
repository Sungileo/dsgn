import pandas  as pd
import numpy as np
from sklearn.model_selection import train_test_split
train = pd.read_csv("insurance/train.csv")
y = train['Response']
X = train.drop(["id", 'Response'],axis=1)

X_train, X_test, y_train, y_test = train_test_split(X, y, train_size=0.8, random_state=24)

X_train['Vehicle_Damage'] = X_train['Vehicle_Damage'].apply(lambda x: 1 if x =="Yes" else 0)
X_test['Vehicle_Damage'] = X_test['Vehicle_Damage'].apply(lambda x: 1 if x =="Yes" else 0)

데이터 전처리

  • Region_code는 값이 너무 많아 제외
  • Driving_License, Previously_Insured, Vehicle_Damage는 0, 1 값을 가지기 때문에 전처리 X
from sklearn.preprocessing import OneHotEncoder, StandardScaler
encoder = OneHotEncoder(sparse_output=False)
train_cat = encoder.fit_transform(X_train[['Gender',"Vehicle_Age"]])
test_cat = encoder.transform(X_test[['Gender', "Vehicle_Age"]])

from sklearn.preprocessing import StandardScaler
scaler = StandardScaler()
train_num = scaler.fit_transform(X_train[["Age", "Annual_Premium", "Policy_Sales_Channel","Vintage"]])
test_num = scaler.transform(X_test[["Age", "Annual_Premium", "Policy_Sales_Channel","Vintage"]])

train_bin = X_train[['Driving_License', "Previously_Insured","Vehicle_Damage"]].values
test_bin = X_test[['Driving_License', "Previously_Insured","Vehicle_Damage"]].values

train_bin = train_bin.astype(np.int32)
test_bin = test_bin.astype(np.int32)

train_prepare = np.concatenate([train_cat, train_num, train_bin], axis=1)
test_prepare = np.concatenate([test_cat, test_num, test_bin], axis=1)
